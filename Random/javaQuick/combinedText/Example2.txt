// Example 2 : Java arrays

class Example2
{
	public static void main(String args[])
	{
		int temp[]={1,2,3,4,5};
		for(int i=0;i<temp.length;i++)
		{
			System.out.print(i+" ");
		}
		System.out.println();
	}
}
